=> REVERSE 
	
	Syntax 

		SELECT REVERSE('Hello World');

		SELECT REVERSE('meow meow');

	----------------------------------------------------------------------

	Exercise 

		SELECT * FROM persons;

		SELECT REVERSE(firstname) FROM persons;

	----------------------------------------------------------------------

	=> REVERSE() with CONCAT()

		SELECT CONCAT('woof',' = ',REVERSE('woof')); 

		SELECT CONCAT(firstname,' = ',REVERSE(firstname)) FROM persons;

	----------------------------------------------------------------------
